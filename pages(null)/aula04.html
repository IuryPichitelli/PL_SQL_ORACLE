FILTROS AVANÇADOS

    Comando Descrição Exemplo
    * Termo para selecionar todas as colunas de uma tabela SELECT * FROM funcionarios;
    SELECT Seleciona a coluna que queremos SELECT * FROM funcionarios;
    FROM Usado para referenciar qual tabela iremos extrair os dados SELECT * FROM funcionarios;
    WHERE Condição (ONDE) que podemos usar para colocar condições na pesquisa SELECT * FROM funcionarios WHERE id = 2;
    LIKE Usado para procurar qualquer sequência sendo ela numerica ou alfabetica SELECT * FROM funcionarios WHERE name_func LIKE 'Pedro%';
    LIKE 'x%' Sequência aonde os caracteres a direita da palavra podem ser qualquer um SELECT * FROM funcionarios WHERE name_func LIKE 'Pedro%';
    LIKE 'x_' Procura por um unico caracter que começe com ele SELECT * FROM funcionarios WHERE name_func LIKE 'Pedro_';
